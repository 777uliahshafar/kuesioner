% handy for likert-style questions
\NewDocumentCommand{\likertquestion}{ O{Strongly disagree} O{Strongly agree} O{7} m }{%
\setcounter{scaleCounter}{0}
\def\ltablecfg{}
\loop\ifnum\thescaleCounter<#3
  \stepcounter{scaleCounter}
  \edef\ltablecfg{%
    \ltablecfg
    c|
  }%
\repeat

\setcounter{scaleCounter}{0}
\def\lheader{}
\loop\ifnum\thescaleCounter<#3
  \stepcounter{scaleCounter}
  \edef\lheader{%
    \lheader
    \arabic{scaleCounter} &
  }%
\repeat

\setcounter{scaleCounter}{0}
\def\lbullets{}
\loop\ifnum\thescaleCounter<#3
  \stepcounter{scaleCounter}
  \edef\lbullets{%
    \lbullets
    $\ocircle$ &
  }%
\repeat

\begin{center}%
#4\\[1em]%
\begin{tabular}{r|\ltablecfg l}%
 & \lheader \\%
\hline%
#1 & \lbullets #2\\
\hline%
\end{tabular}%
\\[2em]
\end{center}%
}
